;
; FLOATE PACKAGE TEST PROGRAM
;

	EXTERN	JTOD
	ExTERN	DTOJ
	EXTERN	DADD
	EXTERN	DSUB
	EXTERN	DMLT
	EXTERN	DDIV
	EXTERN	DCMP
	EXTERN	DMOV

BDOS	EQU	0005H

;	ORG	100H

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,JVAL2
	LD	DE,VAL2
	CALL	JTOD

	LD	HL,VAL2
	LD	DE,TMP1
	CALL	DMOV

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DSUB

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,JVAL2
	LD	DE,VAL2
	CALL	JTOD

	LD	HL,VAL2
	LD	DE,TMP1
	CALL	DMOV

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DMLT

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,JVAL2
	LD	DE,VAL2
	CALL	JTOD

	LD	HL,VAL2
	LD	DE,TMP1
	CALL	DMOV

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DDIV

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	RET

JVAL1:	DEFB	'+3.0D+0 '
JVAL2:	DEFB	'+8.0D+1 '
VAL1:	DEFS	8
VAL2:	DEFS	8
TMP1:	DEFS	8
RES:	DEFS	32
	DEFB	'$'
