BDOS  EQU 0005H

	ORG	100H

	LD	HL,JVAL
	LD	DE,VAL
	CALL	JTOD

	LD	HL,VAL
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS
	RET

JVAL:	DEFB	'1.2345678D+2 '
VAL:	DEFS	8
RES:	DEFS	32
	DEFB	'$'

	ORG	300H
