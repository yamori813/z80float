;
; FLOATE PACKAGE TEST PROGRAM
;

	EXTERN	JTOD
	EXTERN	DTOJ
	EXTERN	DLOG
	EXTERN	DEXP
	EXTERN	DMOV

BDOS	EQU	0005H

;	ORG	100H

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DLOG

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	LD	HL,TMP1
	LD	DE,VAL1
	CALL	DMOV

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DEXP

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	RET

JVAL1:	DEFB	'+2.0D+0 '
VAL1:	DEFS	8
TMP1:	DEFS	8
RES:	DEFS	32
	DEFB	'$'

; 0.69314718055995
