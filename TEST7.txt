;
; FLOATE PACKAGE TEST PROGRAM
;

	EXTERN	JTOd
	EXTERN	DTOJ
	EXTERN	DMOD

	EXTERN	PRINT

;	ORG	100H

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,JVAL2
	LD	DE,TMP1
	CALL	JTOD

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DMOD

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	DE,RES
	CALL	PRINT

	RET

JVAL1:	DEFB	'+9.0D+0 '
JVAL2:	DEFB	'+123.0D+0 '
VAL1:	DEFS	8
TMP1:	DEFS	8
RES:	DEFS	32
	DEFB	'$'
