;
; FLOATE PACKAGE TEST PROGRAM
;

	EXTERN	JTOD
	ExTERN	DTOJ
	EXTERN	DSQRT

BDOS	EQU	0005H

;	ORG	100H

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DSQRT

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	LD	HL,JVAL2
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DSQRT

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	RET

JVAL1:	DEFB	'+2.0D+0 '
JVAL2:	DEFB	'+3.0D+1 '
VAL1:	DEFS	8
TMP1:	DEFS	8
RES:	DEFS	32
	DEFB	'$'
