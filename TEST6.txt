;
; FLOATE PACKAGE TEST PROGRAM
;

	EXTERN	JTOD
	ExTERN	DTOJ
	EXTERN	DSIN
	EXTERN	DCOS
	EXTERN	DATAN

	EXTERN	PRINT
	EXTERN	EXIT

;	ORG	100H

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DSIN

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	DE,RES
	CALL	PRINT

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DCOS

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	DE,RES
	CALL	PRINT

	LD	HL,VAL1
	LD	DE,TMP1
	CALL	DATAN

	LD	HL,TMP1
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	DE,RES
	CALL	PRINT

	JP	EXIT

JVAL1:	DEFB	'+2.0D+0 '
VAL1:	DEFS	8
TMP1:	DEFS	8
RES:	DEFS	32
	DEFB	'$'
