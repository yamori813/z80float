;
; FLOATE PACKAGE TEST PROGRAM
;

	EXTERN	JTOD
	EXTERN	DTOJ

	EXTERN	PRINT
	EXTERN	EXIT

;	ORG	100H

	LD	HL,JVAL
	LD	DE,VAL
	CALL	JTOD

	LD	HL,VAL
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	DE,RES
	CALL	PRINT
	JP	EXIT

JVAL:	DEFB	'1.2345678D+2 '
VAL:	DEFS	8
RES:	DEFS	32
	DEFB	'$'
