;
; FLOATE PACKAGE TEST PROGRAM
;

	EXTERN	JTOD
	ExTERN	DTOJ
	EXTERN	DSIN
	EXTERN	DCOS
	EXTERN	DDIV

BDOS	EQU	0005H

;	ORG	100H

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,VAL1
	LD	DE,SVAL
	CALL	DSIN

	LD	HL,VAL1
	LD	DE,CVAL
	CALL	DCOS

	LD	HL,CVAL
	LD	DE,SVAL
	CALL	DDIV

	LD	HL,SVAL
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	RET

JVAL1:	DEFB	'+1.0D+0 '
VAL1:	DEFS	8
SVAL:	DEFS	8
CVAL:	DEFS	8
RES:	DEFS	32
	DEFB	'$'
