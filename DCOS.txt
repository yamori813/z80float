	TITLE	DCOS (V02/L01)
;
	GLOBAL	DCOS
	EXTERN	DSIN
	EXTERN	DADD
	EXTERN	DMOV
;
DCOS	EQU	$
	LD	(SRCADR),HL
	LD	(DSTADR),DE
	LD	DE,X
	CALL	DMOV
	LD	HL,PAI02
	LD	DE,X
	CALL	DADD
	LD	HL,X
	LD	DE,(DSTADR)
	JP	DSIN
;
SRCADR:	DEFW	0
DSTADR:	DEFW	0
X:	DEFB	0,0,0,0,0,0,0,0
PAI02:	DEFB	1AH,2DH,44H,54H
	DEFB	0FBH,21H,19H,41H
	END
