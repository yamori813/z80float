;
; FLOATE PACKAGE TEST PROGRAM
;

BDOS  EQU 0005H

	ORG	100H

	LD	HL,JVAL1
	LD	DE,VAL1
	CALL	JTOD

	LD	HL,JVAL2
	LD	DE,VAL2
	CALL	JTOD

	LD	HL,VAL1
	LD	DE,VAL2
	CALL	DADD

	LD	HL,VAL2
	LD	DE,RES
	CALL	DTOJ

; DISPLAY RESULT
	LD	C,09H
	LD	DE,RES
	CALL	BDOS

	RET

JVAL1:	DEFB	'1.1111111D+2 '
JVAL2:	DEFB	'2.2222222D+2 '
VAL1:	DEFS	8
VAL2:	DEFS	8
RES:	DEFS	32
	DEFB	'$'

	ORG	300H

include "ASUB1.TXT"
include "ASUB2.TXT"
include "ROUND.TXT"

include "JTOD.TXT"
include "DTOJ.TXT"

include "DFLT.TXT"
